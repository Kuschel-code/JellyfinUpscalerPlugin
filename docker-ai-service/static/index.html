Authentication

All mutating endpoints require the X-Api-Token header when API_TOKEN is set.

# Health (no auth)
curl -s $SERVICE/status | jq .

# Load a model (auth required)
curl -s -X POST $SERVICE/models/load \
  -H "X-Api-Token: $API_TOKEN" \
  -F model_name=realesrgan-x2

# Upscale a frame
curl -s -X POST $SERVICE/upscale-frame \
  -H "X-Api-Token: $API_TOKEN" \
  -F file=@frame.png > out.png
