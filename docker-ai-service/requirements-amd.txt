# AI/ML - ONNX Runtime: do NOT list onnxruntime here.
# The AMD image installs onnxruntime-rocm in Dockerfile.amd, which PROVIDES the
# `onnxruntime` module itself. Installing plain onnxruntime alongside it shadows
# the ROCm build, so the runtime reports AzureExecutionProvider/CPU and the GPU
# is never used (FIX-2 — the real cause of the CPU-only AMD image).
numpy>=1.24.0,<2.0.0

# Image Processing - OpenCV with DNN Super Resolution
# Capped <4.12: opencv 4.12+ requires numpy>=2, but the rocm/pytorch base ships
# torch 2.3 compiled against numpy 1.x (numpy 2 breaks it). CVE-2025-53644
# (Jpeg2K readHeader) therefore stays open HERE until the ROCm base moves to a
# torch>=2.4 image - low risk for this service (frames are PNG/JPEG, no JPEG2000).
opencv-contrib-python>=4.10.0,<4.12.0
